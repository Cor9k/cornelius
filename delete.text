const deleteBClass = "delete-btn";

var caseDivs = document.getElementsByClassName(caseDivClass);

______________________________________INNENFOR______________________________________________
for (let i = 0; i < caseDivs.length; i++) {
	let curCaseDiv = caseDivs[i];

	let children = curCaseDiv.getElementsByTagName('*');

	let deleteB = findFirstOfClass(deleteBClass, children);

	deleteB.addEventListener("click", function () {
		removeCase(curCaseDiv);
	});
}
_______________________________UTENFOR______________________________________________________

function removeCase(caseDiv) {
	caseDiv.parentNode.removeChild(caseDiv);
}
